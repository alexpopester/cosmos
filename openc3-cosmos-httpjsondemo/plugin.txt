# LANGUAGE python
# Set VARIABLEs here to allow variation in your plugin

VARIABLE httpjsondemo_target_name HTTPJSONDEMO
VARIABLE httpjsondemo_host api.example.com
VARIABLE httpjsondemo_port 80
VARIABLE httpjsondemo_protocol http
VARIABLE httpjsondemo_path /api/v1/data
VARIABLE httpjsondemo_method post
VARIABLE httpjsondemo_content_type application/json
VARIABLE httpjsondemo_server_port 4567
# Name of the COSMOS secret holding the inbound API key (set via Admin → Secrets).
# Leave blank to disable authentication.
VARIABLE httpjsondemo_api_key_secret HTTPJSONDEMO_API_KEY
VARIABLE httpjsondemo_max_queue_depth 1000

TARGET HTTPJSONDEMO <%= httpjsondemo_target_name %>
  # Decom tlm and cmd data in the time-series database is kept forever by default.
  # Uncomment to enforce a 30 day retention time. See https://docs.openc3.com/docs/configuration/plugins
  # CMD_DECOM_RETAIN_TIME 30d
  # TLM_DECOM_RETAIN_TIME 30d

# Outbound: COSMOS sends commands as JSON HTTP requests to an external service.
INTERFACE <%= httpjsondemo_target_name %>_INT http_json_client_interface.py <%= httpjsondemo_host %> <%= httpjsondemo_port %> <%= httpjsondemo_protocol %> 5 10 5
  OPTION PATH <%= httpjsondemo_path %>
  OPTION METHOD <%= httpjsondemo_method %>
  OPTION HEADER Content-Type <%= httpjsondemo_content_type %>
  OPTION RESPONSE_PACKET <%= httpjsondemo_target_name %> RESPONSE
  OPTION ERROR_PACKET <%= httpjsondemo_target_name %> ERROR
  MAP_TARGET <%= httpjsondemo_target_name %>

# Inbound: external sources POST JSON to /targetname/packetname and COSMOS stores it as telemetry.
INTERFACE <%= httpjsondemo_target_name %>_SERVER_INT http_json_server_interface.py <%= httpjsondemo_server_port %>
  # Require callers to send X-Api-Key: <value> matching the COSMOS secret below.
  # Remove the SECRET line to disable authentication entirely.
  SECRET ENV <%= httpjsondemo_api_key_secret %> <%= httpjsondemo_api_key_secret %> API_KEY
  OPTION MAX_QUEUE_DEPTH <%= httpjsondemo_max_queue_depth %>
  MAP_TARGET <%= httpjsondemo_target_name %>
